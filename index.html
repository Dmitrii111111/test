Тестовое по работе с Git

Система контроля версий

Git позволяет следить за происходящими изменениями, а также работать над одним заданием одновременно нескольким разработчикам. Например, контроль версий помогает отслеживать исправления в текстовых документах или сохранять версии измененных файлов.

#1 Задаем имя пользователя и email под которым мы будем вносить все изменения в файлы

git config --global user.name "Dima Ushakov"

git config --global user.email "ushako160696@gmail.com"

#2 Нашу директорию превращаем в GIT репозиторий

Открываем нашу директорию в терминале

git init # сообщает гиту о том что эту директорию нужно превратить в репозиторий(следить за файлами и папками которые здесь находятся)

#3 Создаем удаленный репозиторий в github

#4 Проложим связь между локальным репозиторием и удаленным на github

git remote add origin https://github.com/Dmitrii111111/test.git

#5 Чтобы скачать копию удаленного репозитория нужно сначала зайти в директорию куда мы хотим ее скачать

git clone https://github.com/Dmitrii111111/test.git

после того как скачали Git уже будет следить за всеми изменениями и будет автоматически связан с удаленным

#6 Внесение изменений в локальный репозиторий

git status # эта команда выведет информацию о всех не закоммиченных файлах в Git репозитории

git add . # Добавить все измененные файлы для commit

git add <имя_файла> # Добавить только указанный файл для commit

git commit -m "Описания для версии" #описываем чтобы понимать что содержится в этой версии

теперь git запомнил то состояние файлов которое мы закоммитили

если мы изменим файлы то нам нужно будет снова добавить их в Индексированный git add . либо git add <имя_файла>

git commit -m "обновление файла index.html"

git log # показывает информацию об всех commit с хешем(с номером) каждого коммита

git checkout <хеш_комита_к_которому_хотим_вернуться> # вернуться к нужной версии commit

#7 Посылаем из локального репозитория в удаленный

git push <имя_удаленного_репозитория(origin)> <имя_ветки_изменений(master)> # локальный знает имя своего удаленного репозитория под кодовым именем origin # master самая главная наша ветка

#8 Допустим другой участник команды хочет скачать ваши коммиты из удаленного репозитория

git pull <имя_удаленного_репозитория(origin)> <имя_ветки(master)> # скачано не просто последняя версия, еще и все коммиты

#9 Ветки в Git

отклоняетесь от основной линии разработки и продолжаете работу независимо от неё, не вмешиваясь в основную линию(главная по умолчанию master)

git branch # выводит список всех локальных веток(звездочкой отмечена та ветка на которой мы сейчас находимся)

git branch <имя_новой_ветки> # чтобы создать новую ветку из ветки master

git checkout <имя_ветки> # чтобы переключиться на любую ветку

последующие коммиты будут уже созданы в ветке в которую мы перешли

git checkout <имя_ветки> # соответственно версии файлов будут меняться в соответствии с тем состояниям в котором его оставили

каждый разработчик может работать в своих ветках не мешая друг другу

наша ветка локальная

#10 Публикация новой ветки на удаленный репозиторий

git push <имя_удаленного_репозитория(origin)> <имя_новой_ветки_(новая_ветка_в_которую_загрузим_нашу_локальную)> # обычно новую ветку называют также как и вашу-которая локальная

теперь другие участники команды могут видеть мою ветку

#11 Слияние веток

есть два механизма для слияния

#1 merge # слияние происходит с помощью одного нового коммита

допустим мы хотим перенести нашу ветку в ветку master

git checkout master # переключаемся на ветку в которую будем делать слияние

git merge <имя_ветки_из_которой_будем_делать_слияние> # git берет все изменения из передовой ветки и добавляет все их в ветку master в виде всего одного нового коммита

#2 rebase # слияние происходит с помощью всех коммитов которые были в ветке

допустим мы хотим перенести нашу ветку в ветку master

git checkout master # переключаемся на ветку в которую будем делать слияние, то есть в master

git rebase <имя_ветки_из_которой_будем_делать_слияние> # git берет все изменения из передовой ветки и добавляет все их в ветку master в виде всех коммитов которые были сделаны в ветке из которой сделали слияние
